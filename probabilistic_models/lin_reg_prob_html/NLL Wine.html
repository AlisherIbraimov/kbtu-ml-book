#   Define a function to calculate the log likelihood required for the Guassian aproach

def calcLogLikelihood(guess, true, n):
    error = true - guess
    sigma = np.std(error)
    f = ((1.0/(2.0*math.pi*sigma*sigma))**(n/2))*np.exp(-1*((np.dot(error.T, error))/(2*sigma*sigma)))
    return np.log(f)

#   Define my function which will return the objective function to be minimized

def myFunction(var):
    [x, y] = [df['fixed acidity'],df['pH']]
    yGuess = (var[2]*(x**2)) + (var[1]*x) + var[0]
    f = calcLogLikelihood(yGuess, y, float(len(yGuess)))
    return (-1*f)

from scipy.optimize import minimize
res = minimize(myFunction, var, method='BFGS',

#   perform least squares fit using scikitlearn

from sklearn.preprocessing import PolynomialFeatures
from sklearn.linear_model import LinearRegression
from sklearn.pipeline import Pipeline
model = Pipeline([('poly', PolynomialFeatures(degree=2)),
    ('linear', LinearRegression(fit_intercept=False))])

#   Here we need to transform fixed acidity column into 1-dimensional list required for the fuction to work properly
#   Also, here we are making the predictions using our model by trying to find y_pred

model = model.fit(df['fixed acidity'].values.reshape(-1, 1), df['pH'])
coefs = model.named_steps['linear'].coef_
y_pred = model.predict(df['fixed acidity'].values.reshape(-1, 1))

#   let's see the metrics for test data

print("MSE:",np.mean((y_pred - df['pH']) ** 2))
